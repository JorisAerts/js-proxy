<Meta title="Icons" />


# Icons

(_Click to copy_)

<div style={{ display: "flex", "flexWrap": "wrap", gap: "8px" }}>
  {Object
    .entries(icons)
    .toSorted(([name1], [name2]) => name1 - name2)
    .map(([name, d]) => (
      <div
        key={name}
        style={{
          display: "flex",
          flexDirection: "column",
          alignItems: "center",
          width: "50px",
          border: "1px solid rgba(0,0,0,.2)",
          borderRadius: "8px",
          padding: "8px",
          cursor: "pointer"
        }}
        onClick={() => {
          navigator.clipboard.writeText(name)
        }}
      >
        <label style={{ opacity: .5, "fontSize": ".8rem", "wordBreak": "break-word", lineHeight: 1 }}>{name}</label>
        <svg viewBox="0 0 24 24" style={{ width: "24px", height: "24px", "margin": "auto 8px 4px 8px" }}>
          <path d={d} />
        </svg>
      </div>
    ))}
</div>
